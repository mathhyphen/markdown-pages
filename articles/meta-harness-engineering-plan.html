## 你的角色

- 如果涉及敏感信息：遵循安全准则

class PromptManager:
    """System Prompt 的管理器 - 支持动态调整"""
    
    def __init__(self, base_prompt=BASE_SYSTEM_PROMPT):
        self.base_prompt = base_prompt
        self.custom_rules = []
        self.output_format = None
    
    def add_rule(self, rule: str):
        """添加自定义规则"""
        self.custom_rules.append(rule)
    
    def set_output_format(self, format_template: str):
        """设置输出格式"""
        self.output_format = format_template
    
    def build(self, company_name: str, context: str = "") -> str:
        """构建最终的 system prompt"""
        prompt = self.base_prompt.format(company_name=company_name)
        
        if self.custom_rules:
            prompt += "\n## 额外规则\n"
            for i, rule in enumerate(self.custom_rules, 1):
                prompt += f"{i}. {rule}\n"
        
        if self.output_format:
            prompt += f"\n## 特定输出格式\n{self.output_format}"
        
        if context:
            prompt += f"\n## 当前上下文\n{context}"
        
        return prompt

class Guide:
    """前馈控制器 - 在行动前引导"""
    
    def __init__(self):
        self.rules: List[Callable] = []
    
    def add_rule(self, rule_func: Callable):
        """添加引导规则"""
        self.rules.append(rule_func)
    
    def check(self, query: str) -> Optional[str]:
        """执行所有规则，返回第一个触发的干预消息"""
        for rule in self.rules:
            intervention = rule(query)
            if intervention:
                return intervention
        return None

class Sensor:
    """反馈传感器 - 在行动后检测"""
    
    def __init__(self):
        self.checks: List[Callable] = []
    
    def add_check(self, check_func: Callable):
        """添加检测项"""
        self.checks.append(check_func)
    
    def inspect(self, answer: str) -> List[str]:
        """执行所有检测，返回发现的问题列表"""
        issues = []
        for check in self.checks:
            found_issues = check(answer)
            if found_issues:
                issues.extend(found_issues)
        return issues

    def create_off_topic_guide(topics: List[str]):
        """检测离题查询"""
        def check(query: str) -> Optional[str]:
            # 简化版：实际可以使用分类器
            for topic in topics:
                if topic.lower() in query.lower():
                    return None  # 相关话题
            return ("您的问题似乎超出了我的知识范围。"
                   "我主要可以帮您解答以下领域的问题：" + 
                   ", ".join(topics[:5]))
        return check
    
    # ========== 内置的 Sensor 检测 ==========
    
    @staticmethod
    def create_citation_sensor():
        """检测是否缺少引用"""
        def check(answer: str) -> List[str]:
            issues = []
            if "参考来源" not in answer and "来源" not in answer:
                issues.append("缺少引用来源")
            return issues
        return check
    
    @staticmethod
    def create_length_sensor(max_chars: int = 1000):
        """检测回答长度"""
        def check(answer: str) -> List[str]:
            issues = []
            if len(answer) > max_chars:
                issues.append(f"回答过长 ({len(answer)}字符，建议<{max_chars})")
            elif len(answer) < 50:
                issues.append("回答过短，可能不够详细")
            return issues
        return check
    
    @staticmethod
    def create_confidence_sensor():
        """检测不确定的表达"""
        def check(answer: str) -> List[str]:
            issues = []
            uncertain_phrases = ["可能", "也许", "大概", "我不确定"]
            for phrase in uncertain_phrases:
                if phrase in answer:
                    issues.append(f"包含不确定表达：'{phrase}'")
            return issues
        return check
    
    def process(self, query: str, answer: str) -> FeedbackResult:
        """
        处理完整的反馈循环
        """
        # 阶段1: 前馈检查
        guide_intervention = self.guide.check(query)
        if guide_intervention:
            return FeedbackResult(
                status=FeedbackStatus.GUIDED,
                message=guide_intervention
            )
        
        # 阶段2: 反馈检测
        issues = self.sensor.inspect(answer)
        
        if issues:
            # 尝试自动修复
            suggestion = self._generate_suggestion(issues)
            
            return FeedbackResult(
                status=FeedbackStatus.NEEDS_REVISION,
                issues=issues,
                suggestion=suggestion
            )
        
        return FeedbackResult(status=FeedbackStatus.OK, message=answer)
    
    def _generate_suggestion(self, issues: List[str]) -> str:
        """根据问题生成修复建议"""
        suggestions = {
            "缺少引用来源": "请在回答末尾添加参考来源",
            "回答过长": "请精简回答，保留核心信息",
            "回答过短": "请补充更多细节和说明",
            "包含不确定表达": "如果不确定，请明确说明或建议查阅其他资料"
        }
        
        fixes = [suggestions.get(issue, "请重新审视回答") for issue in issues]
        return "；".join(fixes)

            "total_candidates": len(candidates),
            "best_score": index["best_score"],
            "avg_score": sum(scores) / len(scores) if scores else 0,
            "improvement": (
                scores[-1] - scores[0] 
                if len(scores) >= 2 else 0

        """
        在历史中搜索
        模拟 grep 功能！
        """
        results = []
        
        for root, dirs, files in os.walk(self.base_dir):
            # 跳过索引文件
            if "index.json" in files:
                continue
            
            for file in files:
                if file.endswith('.py') or file.endswith('.json'):
                    filepath = os.path.join(root, file)
                    try:
                        with open(filepath, 'r') as f:
                            content = f.read()
                            if pattern.lower() in content.lower():
                                results.append({
                                    "file": filepath,
                                    "matches": content.count(pattern)
                                })
                    except:
                        pass
        
        return sorted(results, key=lambda x: x["matches"], reverse=True)

import os

from typing import Dict, List, Optional, Any
from dataclasses import dataclass

@dataclass
class Proposal:
    """提出的候选配置"""
    candidate_id: int
    config: Dict[str, Any]

    source_code: str  # 生成的 Harness 代码

class ProposerAgent:
    """
    Proposer Agent - Meta-Harness 的核心！
    
    它是一个编码 Agent，能够：
    1. 通过文件系统访问所有历史候选
    2. 选择性地读取相关历史（模拟 grep/cat）
    3. 分析成功/失败的模式
    4. 提出新的、改进的配置
    """
    
    def __init__(
        self, 
        model: str = "claude-sonnet-4-20250514",
        history: FilesystemHistory = None
    ):
        self.model = model
        self.llm = OpenAI()  # 或 Anthropic client
        self.history = history or FilesystemHistory()
        
        self.system_prompt = """你是一个 Harness Optimization Expert（Harness 优化专家）。
        
## 你的任务
基于历史优化记录，提出新的 Harness 配置以提高系统性能。

## 你可以使用的工具
- read_file(path): 读取文件内容
- grep(pattern, dir): 在目录中搜索模式
- get_best(): 获取最佳候选
- get_recent(n): 获取最近N个候选
- get_stats(): 获取历史统计

## 工作流程
1. 分析历史趋势：哪些配置效果最好？
2. 识别失败模式：什么导致了低分？
3. 形成假设：如何改进？
4. 提出新配置：具体的参数调整
5. 生成理由：为什么这样改？

## 输出格式
返回 JSON：
{
    "config": {...},
    "reasoning": "详细说明",
    "source_code": "生成的代码"
}

## 重要原则
- 不要盲目随机搜索，要有依据地调整
- 关注失败案例中的模式
- 小步迭代，避免剧烈变化
- 记录你的推理过程
"""
    
    def propose(self, iteration: int) -> Proposal:
        """
        提出一个新的候选配置
        """
        print(f"\n🤖 Proposer 正在工作 (Iteration {iteration})...")
        
        # Step 1: 收集上下文
        context = self._gather_context()
        
        # Step 2: 调用 LLM 生成提案
        proposal = self._generate_proposal(context, iteration)
        
        return proposal
    
    def _gather_context(self) -> str:
        """收集相关历史上下文"""
        context_parts = []
        
        # 1. 获取统计信息
        stats = self.history.get_history_stats()
        context_parts.append(f"## 历史统计\n```json\n{json.dumps(stats, indent=2)}\n```")
        
        # 2. 获取最佳候选

        if best:
            context_parts.append(f"""

```json
{json.dumps(best.config, indent=2)}
```

### 分数明细
{json.dumps(best.score_breakdown, indent=2)}

### 元数据
{json.dumps(best.metadata, indent=2)}
""")
        
        # 3. 获取最近的几个候选（用于观察趋势）
        recent = self.history.get_recent_candidates(n=5)

            context_parts.append("\n## 最近5个候选的趋势\n")

{context}

基于以上历史数据，分析趋势并提出新的 Harness 配置。

1. 如何提高准确率？

"""}

            model=self.model,

        )
        
        result = json.loads(response.choices[0].message.content)
        
        return Proposal(
            candidate_id=iteration,
            config=result.get("config", {}),
            reasoning=result.get("reasoning", ""),
            source_code=result.get("source_code", "")
        )

"""
        
        response = self.llm.chat.completions.create(

⚡ 常见陷阱预警

  • ❌ 陷阱1：过度准备综合症
    花了太多时间看视频、读文章，却迟迟不动手。
    ✅ 解药：给自己设定"第一天就必须写出能运行的代码"的硬性目标。
  • ❌ 陷阱2：完美主义陷阱
    想一次性做出完美的系统，导致进度停滞。
    ✅ 解药：接受"MVP思维"——先做一个能用的烂版本，再迭代改进。
  • ❌ 陷阱3：孤立无援
    一个人死磕，遇到问题卡很久。
    ✅ 解药：加入 Discord/Slack 社区，积极参与讨论，善用 AI 助手辅助调试。
  • ❌ 陷阱4：忽视评估
    只关注功能实现，不建立量化评估体系。
    ✅ 解药：从第一天就开始收集指标，数据驱动优化。
